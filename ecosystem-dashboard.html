rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    function isAuth() { return request.auth != null; }

    // Ecosystem users — hub SSO
    match /users/{uid} {
      allow read, write: if isAuth();
    }
    // Notifikasi lintas portal
    match /notifications/{id} {
      allow read, write: if isAuth();
    }
    // Statistik agregat
    match /ecosystem_stats/{id} {
      allow read: if true;
      allow write: if isAuth();
    }
  }
}

📌 PROJECT: hidayatulamin-e6f22 (Pendidikan + SGT)
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    function isAuth() { return request.auth != null; }
    function isRole(role) {
      return isAuth() &&
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == role;
    }

    match /users/{uid} {
      allow read: if isAuth();
      allow write: if isAuth();
    }
    match /sgt_transactions/{id} {
      allow read: if isAuth();
      allow create: if isAuth();
      allow update, delete: if isRole('admin');
    }
    match /sgt_config/{id} {
      allow read: if isAuth();
      allow write: if isRole('admin');
    }
    match /donasi/{id} {
      allow read: if true;
      allow create: if true;
      allow update: if isRole('admin');
    }
    match /hafalan/{id} {
      allow read: if isAuth();
      allow write: if isAuth();
    }
    match /{document=**} {
      allow read, write: if isAuth();
    }
  }
}
